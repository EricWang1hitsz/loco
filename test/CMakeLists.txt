add_subdirectory(gait_pattern EXCLUDE_FROM_ALL)
add_subdirectory(limb_coordinator EXCLUDE_FROM_ALL)
add_subdirectory(torso_control EXCLUDE_FROM_ALL)
add_subdirectory(locomotion_controller EXCLUDE_FROM_ALL)
